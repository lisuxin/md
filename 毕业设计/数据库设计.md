学生表：`gl_classandgrade_student`

```mysql
create table gl_classandgrade_student(
    studentnumber bigint(20) primary key auto_increment,                                             -- 学号   组成：学校标识（3）+年级标识（3）自己id：（5）一共12位
    name char(50),                                                      -- 姓名
    password char(20),                                                  -- 密码
    sex nvarchar(2) CHECK(gl_classandgrade_student.sex=N'男' or gl_classandgrade_student.sex=N'女'),      -- 性别
    idnamber char(30) unique,                                           -- 身份证号
    phonenamber char(30),                                               -- 电话号码
    address char(100),                                                  -- 住址
    grade_id int(10) ,                          -- 年级
    classandgrade_id int(10)        -- 班级
);

-- 添加外键
ALTER TABLE glwc.`gl_classandgrade_student` ADD FOREIGN KEY (grade_id) REFERENCES gl_grade (id);
ALTER TABLE glwc.`gl_classandgrade_student` ADD FOREIGN KEY (classandgrade_id) REFERENCES gl_classandgrade (id);
```

教师表：`gl_teacher`

```mysql
create table gl_teacher(
    jobnumber int(20)  primary key auto_increment, -- 工号 组成：学校标识（3）+自己id(6)
    name char(50),                   -- 姓名
    password char(20),                                                  -- 密码
    sex nvarchar(2) CHECK(gl_teacher.sex=N'男' or gl_teacher.sex=N'女'),      -- 性别
    idnamber char(30) unique,       -- 身份证号
    phonenamber char(30),            -- 电话号码
    address char(100),                -- 住址
    position_id int(10)               -- 职位
);
ALTER TABLE glwc.`gl_teacher` ADD FOREIGN KEY (position_id) REFERENCES gl_position (id);
```

选择题：`gl_course_choose`

```mysql
create table gl_course_choose(
    id int(10) primary key auto_increment,
    choose  varchar(50) unique, -- 题
    yi char(10),    -- 答案1 正确答案
    er char(10),    -- 答案2
```
